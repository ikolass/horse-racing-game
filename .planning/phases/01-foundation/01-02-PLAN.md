**Step 3:** Create `src/store/modules/schedule.js` — a namespaced Vuex module.

State:
- `rounds`: array of round objects, initially empty `[]`

Each round object shape: `{ roundNumber: Number, distance: Number, horseIndices: Number[] }`

Mutations:
- `SET_ROUNDS(state, rounds)` — sets `state.rounds = rounds`
- `CLEAR_SCHEDULE(state)` — sets `state.rounds = []`

Actions:
- `generateSchedule({ commit, rootState })` — creates 6 rounds:
  1. Import `GAME_CONFIG` from `src/config/gameConfig.js` and `shuffle` from `src/utils/shuffle.js`
  2. Get the horse list from `rootState.horses.list`
  3. For each round (0 to TOTAL_ROUNDS-1):
     - Create array of all horse indices [0..19]
     - Shuffle the indices
     - Take first HORSES_PER_ROUND (10) indices
     - Build round object: `{ roundNumber: i + 1, distance: ROUND_DISTANCES[i], horseIndices: selectedIndices }`
  4. Commit `SET_ROUNDS` with the 6 round objects

Getters:
- `allRounds`: returns `state.rounds`
- `roundByNumber`: returns `(num) => state.rounds.find(r => r.roundNumber === num)`

```js
import { GAME_CONFIG } from '@/config/gameConfig.js';
import { shuffle } from '@/utils/shuffle.js';

const schedule = {
  namespaced: true,
  state: () => ({ rounds: [] }),
  mutations: {
    SET_ROUNDS(state, rounds) { state.rounds = rounds; },
    CLEAR_SCHEDULE(state) { state.rounds = []; },
  },
  actions: {
    generateSchedule({ commit, rootState }) {
      const allIndices = rootState.horses.list.map((_, i) => i);
      const rounds = [];
      for (let i = 0; i < GAME_CONFIG.TOTAL_ROUNDS; i++) {
        const shuffled = shuffle(allIndices);
        const horseIndices = shuffled.slice(0, GAME_CONFIG.HORSES_PER_ROUND);
        rounds.push({
          roundNumber: i + 1,
          distance: GAME_CONFIG.ROUND_DISTANCES[i],
          horseIndices,
        });
      }
      commit('SET_ROUNDS', rounds);
    },
  },
  getters: {
    allRounds: (state) => state.rounds,
    roundByNumber: (state) => (num) => state.rounds.find(r => r.roundNumber === num),
  },
};
export default schedule;
```

**Step 1:** Create `src/store/modules/race.js` — namespaced Vuex module implementing the gamePhase state machine.

State:
- `gamePhase`: string, initial value `'IDLE'` — valid values: `IDLE`, `SCHEDULED`, `RACING`, `ROUND_COMPLETE`, `DONE`
- `currentRound`: number, initial value `0` (no round in progress)
- `positions`: object, initial value `{}` — keys are horse indices, values are progress 0-100 (used in Phase 3 animation, initialized here as empty)
- `intervalId`: number or null, initial value `null` (will hold setInterval ID in Phase 3)

Mutations:
- `SET_GAME_PHASE(state, phase)` — sets `state.gamePhase = phase`
- `SET_CURRENT_ROUND(state, roundNumber)` — sets `state.currentRound = roundNumber`
- `SET_POSITIONS(state, positions)` — sets `state.positions = positions`
- `RESET_POSITIONS(state)` — sets `state.positions = {}`
- `SET_INTERVAL_ID(state, id)` — sets `state.intervalId = id`

Actions:
- `transitionTo({ commit, state }, phase)` — validates the transition is legal, then commits SET_GAME_PHASE. Legal transitions:
  - IDLE -> SCHEDULED (after generate)
  - SCHEDULED -> RACING (after start)
  - RACING -> ROUND_COMPLETE (after a round finishes)
  - ROUND_COMPLETE -> RACING (next round starts)
  - ROUND_COMPLETE -> DONE (last round completed)
  - DONE -> IDLE (reset for new game)
  - If illegal transition, log warning: `console.warn('Invalid transition: ' + state.gamePhase + ' -> ' + phase)`

- `startRace({ commit, dispatch, rootState })` — orchestration action (Phase 3 will expand this):
  1. Dispatch `transitionTo` with `'RACING'`
  2. Set currentRound to 1
  (Phase 3 will add the tick loop here)

Getters:
- `gamePhase`: returns `state.gamePhase`
- `currentRound`: returns `state.currentRound`
- `isIdle`: returns `state.gamePhase === 'IDLE'`
- `isScheduled`: returns `state.gamePhase === 'SCHEDULED'`
- `isRacing`: returns `state.gamePhase === 'RACING'`
- `isDone`: returns `state.gamePhase === 'DONE'`

```js
const VALID_TRANSITIONS = {
  IDLE: ['SCHEDULED'],
  SCHEDULED: ['RACING'],
  RACING: ['ROUND_COMPLETE'],
  ROUND_COMPLETE: ['RACING', 'DONE'],
  DONE: ['IDLE'],
};

const race = {
  namespaced: true,
  state: () => ({
    gamePhase: 'IDLE',
    currentRound: 0,
    positions: {},
    intervalId: null,
  }),
  mutations: {
    SET_GAME_PHASE(state, phase) { state.gamePhase = phase; },
    SET_CURRENT_ROUND(state, roundNumber) { state.currentRound = roundNumber; },
    SET_POSITIONS(state, positions) { state.positions = positions; },
    RESET_POSITIONS(state) { state.positions = {}; },
    SET_INTERVAL_ID(state, id) { state.intervalId = id; },
  },
  actions: {
    transitionTo({ commit, state }, phase) {
      const allowed = VALID_TRANSITIONS[state.gamePhase];
      if (!allowed || !allowed.includes(phase)) {
        console.warn(`Invalid transition: ${state.gamePhase} -> ${phase}`);
        return false;
      }
      commit('SET_GAME_PHASE', phase);
      return true;
    },
    startRace({ commit, dispatch }) {
      dispatch('transitionTo', 'RACING');
      commit('SET_CURRENT_ROUND', 1);
    },
  },
  getters: {
    gamePhase: (state) => state.gamePhase,
    currentRound: (state) => state.currentRound,
    isIdle: (state) => state.gamePhase === 'IDLE',
    isScheduled: (state) => state.gamePhase === 'SCHEDULED',
    isRacing: (state) => state.gamePhase === 'RACING',
    isDone: (state) => state.gamePhase === 'DONE',
  },
};
export default race;
```
