SCRIPT SETUP section:

  import { computed } from 'vue'
  import { useStore } from 'vuex'

  const store = useStore()

  const gamePhase = computed(() => store.getters['race/gamePhase'])
  const currentRound = computed(() => store.getters['race/currentRound'])
  const positions = computed(() => store.state.race.positions)
  const allHorses = computed(() => store.getters['horses/allHorses'])

  // The round to display: current round if set, otherwise round 1 (for SCHEDULED state)
  const visibleRound = computed(() => {
    const roundNum = currentRound.value || 1
    return store.getters['schedule/roundByNumber'](roundNum) ?? null
  })

  // Map horse indices to full horse objects with their index attached
  const racingHorses = computed(() => {
    if (!visibleRound.value) return []
    return visibleRound.value.horseIndices.map(idx => ({
      idx,
      ...allHorses.value[idx],
    }))
  })

  // Convert 0.0-1.0 position to CSS percentage string
  function positionPercent(horseIdx) {
    return ((positions.value[horseIdx] ?? 0) * 100) + '%'
  }

  // Whether a schedule exists (controls empty state vs track display)
  const hasSchedule = computed(() => !!visibleRound.value)


TEMPLATE section structure:

  Root div with data-testid="race-track" and class="race-track"

  IF hasSchedule is true, show the track:
    1. Round header div:
       - class="round-header"
       - data-testid="round-header"
       - Text content: "Round {{ visibleRound.roundNumber }} -- {{ visibleRound.distance }}m"
       - Use an em-dash character (Unicode 2014), not double hyphens

    2. Track container div (class="track-container"):
       - v-for over racingHorses, key is horse.idx
       - Each iteration renders a lane div:
         - class="lane"
         - :class with lane-odd on odd indices (i % 2 === 0 since 0-indexed) and lane-even on even
         - :data-testid="`lane-${horse.idx}`"

         Inside each lane:
           a) Lane label div (class="lane-label"):
              - Swatch span: class="swatch", :style="{ backgroundColor: horse.color }"
              - Name span: class="lane-name", text {{ horse.name }}

           b) Track area div (class="track-area"):
              - Finish line div: class="finish-line", data-testid="finish-line"
              - Horse marker div:
                - class="horse-marker"
                - :data-testid="`marker-${horse.idx}`"
                - :style="{ left: positionPercent(horse.idx) }"
                - Inside marker:
                  - Swatch span: class="marker-swatch", :style="{ backgroundColor: horse.color }"
                  - Name span: class="marker-name", text {{ horse.name }}
                  - Badge span: class="marker-badge", text {{ horse.condition }}

  ELSE (no schedule / IDLE state), show empty state:
    - div class="empty-state"
    - h3 text: "Waiting for schedule..."  (uses --font-heading)
    - p text: "Click Generate Schedule to set up a race."
