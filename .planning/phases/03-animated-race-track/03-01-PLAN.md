(3) runRound({ commit, state, rootState, dispatch }, { horseIndices }) -- NEW action:
  - Defensively clear any existing interval: if (_intervalRef) { clearInterval(_intervalRef); _intervalRef = null }
  - Map horse indices to horse objects: const horses = horseIndices.map(idx => rootState.horses.list[idx])
  - Find max condition: const maxCondition = Math.max(...horses.map(h => h.condition))
  - Pre-compute constant speeds (per D-03, no per-tick randomness):
      const speeds = {}
      horseIndices.forEach((idx, i) => {
        speeds[idx] = horses[i].condition / (maxCondition * GAME_CONFIG.TICKS_TO_WIN)
      })
  - Initialize positions to 0:
      const initPos = {}
      horseIndices.forEach(idx => { initPos[idx] = 0 })
      commit('SET_POSITIONS', initPos)
  - Start the setInterval at GAME_CONFIG.TICK_INTERVAL_MS (50ms). Each tick:
      (a) Spread current positions: const current = { ...state.positions }
      (b) Track allDone = true
      (c) For each horseIdx: if current[idx] < 1.0, add speeds[idx], clamp to 1.0 via Math.min. If still < 1.0, set allDone = false.
      (d) commit('SET_POSITIONS', current)
      (e) If allDone: clearInterval, set _intervalRef = null, commit('SET_INTERVAL_ID', null), dispatch('onRoundComplete')
  - After setInterval: commit('SET_INTERVAL_ID', _intervalRef)

(4) onRoundComplete({ commit, state, dispatch }) -- NEW action:
  - dispatch('transitionTo', 'ROUND_COMPLETE')
  - setTimeout with GAME_CONFIG.PAUSE_BETWEEN_ROUNDS_MS (1500):
      If state.currentRound >= GAME_CONFIG.TOTAL_ROUNDS:
        dispatch('transitionTo', 'DONE')   (D-14: no auto-advance after round 6)
      Else:
        const nextRound = state.currentRound + 1
        commit('RESET_POSITIONS')
        commit('SET_CURRENT_ROUND', nextRound)
        dispatch('transitionTo', 'RACING')
        dispatch('startRoundByNumber', nextRound)

(5) startRoundByNumber({ dispatch, rootGetters }, roundNumber) -- NEW action:
  - const round = rootGetters['schedule/roundByNumber'](roundNumber)
